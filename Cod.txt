        private async void BtnGetWeather_Click(object sender, EventArgs e)
        {
            string city = txtCity.Text.Trim();
            if (string.IsNullOrEmpty(city))
            {
                MessageBox.Show("Введiть назву мiста", "Помилка", MessageBoxButtons.OK, MessageBoxIcon.Warning);
                return;
            }

            btnGetWeather.Enabled = false;
            lblStatus.Text = "Завантаження погоди...";
            rtbResult.Clear();

            string weatherInfo = await GetWeatherAsync(city);
            rtbResult.Text = weatherInfo;

            lblStatus.Text = "Готово";
            btnGetWeather.Enabled = true;
        }

        private async void BtnGetCurrency_Click(object sender, EventArgs e)

        }

        private async Task<string> GetWeatherAsync(string city)
        {
            try
            {
                using (HttpClient client = new HttpClient())
                {
                    string geoUrl = $"https://geocoding-api.open-meteo.com/v1/search?name={city}&count=1&language=uk&format=json";
                    HttpResponseMessage geoResponse = await client.GetAsync(geoUrl);
                    string geoJson = await geoResponse.Content.ReadAsStringAsync();
                    JObject geoData = JObject.Parse(geoJson);

                    if (geoData["results"] == null || !geoData["results"].HasValues)
                    {
                        return $"Мiсто '{city}' не знайдено.\n\nСпробуйте такi мiста Закарпаття:\n- Uzhhorod\n- Mukachevo\n- Berehove\n- Khust\n- Chop\n- Svalyava\n- Rakhiv";
                    }

                    double lat = (double)geoData["results"][0]["latitude"];
                    double lon = (double)geoData["results"][0]["longitude"];
                    string cityName = geoData["results"][0]["name"].ToString();
                    string country = geoData["results"][0]["country"]?.ToString() ?? "";

                    string weatherUrl = $"https://api.open-meteo.com/v1/forecast?latitude={lat}&longitude={lon}&current_weather=true&temperature_unit=celsius&windspeed_unit=ms&timezone=auto";
                    HttpResponseMessage weatherResponse = await client.GetAsync(weatherUrl);
                    string weatherJson = await weatherResponse.Content.ReadAsStringAsync();
                    JObject weatherData = JObject.Parse(weatherJson);

                    double temp = (double)weatherData["current_weather"]["temperature"];
                    double windSpeed = (double)weatherData["current_weather"]["windspeed"];
                    int weatherCode = (int)weatherData["current_weather"]["weathercode"];

                    string description = GetWeatherDescription(weatherCode);

                    return $"ПОГОДА В ЗАКАРПАТТI\n" +
                           $"================================\n" +
                           $"Мiсто: {cityName}\n" +
                           $"Країна: {country}\n" +
                           $"{description}\n" +
                           $"Температура: {temp}°C\n" +
                           $"Вiтер: {windSpeed} м/с\n" +
                           $"================================\n" +
                           $"Закарпаття — край легенд та квiтучих садiв!\n" +
                           $"{DateTime.Now:dd.MM.yyyy HH:mm}";
                }
            }
            catch (Exception ex)
            {
                return $"Помилка: {ex.Message}\n\nПеревiрте пiдключення до iнтернету.";
            }
        }

        private async Task<string> GetCurrencyRatesToUAHAsync()
        {
            try
            {
                using (HttpClient client = new HttpClient())
                {
                    string url = "https://api.exchangerate-api.com/v4/latest/UAH";
                    HttpResponseMessage response = await client.GetAsync(url);

                    if (!response.IsSuccessStatusCode)
                        return $"Помилка отримання курсiв валют: {response.StatusCode}";

                    string json = await response.Content.ReadAsStringAsync();
                    JObject data = JObject.Parse(json);

                    double usdRate = (double)data["rates"]["USD"];
                    double eurRate = (double)data["rates"]["EUR"];
                    double plnRate = (double)data["rates"]["PLN"];
                    double gbpRate = (double)data["rates"]["GBP"];

                    double uahPerUsd = 1 / usdRate;
                    double uahPerEur = 1 / eurRate;
                    double uahPerPln = 1 / plnRate;
                    double uahPerGbp = 1 / gbpRate;

                    return $"КУРСИ ВАЛЮТ ДО ГРИВНI (UAH)\n" +
                           $"================================\n" +
                           $"1 USD = {uahPerUsd:F2} UAH\n" +
                           $"1 EUR = {uahPerEur:F2} UAH\n" +
                           $"1 PLN = {uahPerPln:F2} UAH\n" +
                           $"1 GBP = {uahPerGbp:F2} UAH\n" +
                           $"================================\n" +
                           $"{DateTime.Now:dd.MM.yyyy HH:mm}\n" +
                           $"Джерело: exchangerate-api.com\n" +
                           $"Слава Українi!";
                }
            }
            catch (Exception ex)
            {
                return $"Помилка: {ex.Message}\n\nПеревiрте пiдключення до iнтернету.";
            }
        }
